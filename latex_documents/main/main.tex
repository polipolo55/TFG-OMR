% ── TODO helper ────────────────────────────────────────────────────────────────
% Usage: \todo{What needs to be written here.}
% \long allows multi-paragraph notes. Empty test uses \detokenize so that
% fragile commands like \Cref{} inside the note don't trip up the test.
% No \textit{} wrap on #1 (also fragile around \Cref).
\long\def\todo#1{%
  \begingroup
    \color{orange}\bfseries
    [TODO\ifx\relax\detokenize{#1}\relax\else: #1\fi]%
  \endgroup
}

%     visual heaviness of a full \midrule.

%     long-text columns that need to break across lines.
\newcommand{\tfgrowsep}{\arrayrulecolor{black!25}\midrule\arrayrulecolor{black}}
