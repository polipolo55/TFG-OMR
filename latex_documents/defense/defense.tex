% ── 3. El problema ─────────────────────────────────────────────────────
\begin{frame}{El problema: partitures només en paper o imatge}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \begin{itemize}
      \setlength\itemsep{0.5em}
      \item Molta música notada només existeix \textbf{en paper} o escanejada
      \item Centenars d'estàndards de jazz, a tots els faristols
      \item Els PDF són \textbf{només imatge}\\
            \small\color{muted} no es poden cercar, editar ni reproduir
    \end{itemize}

    \column{0.5\textwidth}
    Digitalitzar-la desbloquejaria:
    \begin{itemize}
      \setlength\itemsep{0.5em}
      \item \textbf{Transposició automàtica} (B\flatt, E\flatt, F)
      \item Reproducció MIDI
      \item Anàlisi harmònica de corpus
      \item Cerca i catalogació
    \end{itemize}
  \end{columns}

  \vspace{1.0em}
  \centering
  {\setlength{\fboxsep}{0pt}\setlength{\fboxrule}{0.5pt}%
   \fcolorbox{muted}{white}{\includegraphics[width=0.6\textwidth]{def_paper}}}
\end{frame}

% ── 4. Quatre reptes ───────────────────────────────────────────────────
\begin{frame}{Quatre reptes}
  \vspace{0.3em}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{\color{accent}1 · Notació bidimensional}\\
    Altura = posició relativa a la clau; durada = cap + plica + corxets.
    \vspace{1.1em}

    \textbf{\color{accent}3 · Escassetat de dades}\\
    Cap corpus etiquetat de \textit{lead sheets} → cal generar dades sintètiques.

    \column{0.48\textwidth}
    \textbf{\color{accent}2 · Bretxa de domini}\\
    Entrenat en net, falla en escanejos reals: soroll, perspectiva, tinta.
    \vspace{1.1em}

    \textbf{\color{accent}4 · Doble flux}\\
    Melodia (pentagrama) + acords (text). L'estat de l'art només fa el primer.
  \end{columns}
\end{frame}

% ── 6. Decisions de disseny ────────────────────────────────────────────
\begin{frame}{Decisions de disseny}
  \begin{columns}[T]
    \column{0.52\textwidth}
    \textbf{Per què CRNN--CTC?}
    \vspace{0.6em}

    {\renewcommand{\arraystretch}{1.35}
    \begin{tabular}{@{}>{\raggedright\arraybackslash}p{2.0cm} >{\raggedright\arraybackslash}p{3.6cm}@{}}
      \toprule
      \textbf{\small Alternativa} & \textbf{\small Motiu de rebuig} \\
      \midrule
      {\small Pipeline modular} & {\small Cal segmentació per símbol} \\
      {\small Transformer (SMT, LEGATO)} & {\small Millor SER, però $>$10× còmput} \\
      {\small Seq2seq + atenció} & {\small Sobrecàrrega sense benefici} \\
      \bottomrule
    \end{tabular}}

    \vspace{0.8em}
    {\small \textbf{Backbone:} ResNet18 — meitat de temps que VGG.}

    \column{0.44\textwidth}
    \textbf{Per què LMX?}
    \vspace{0.5em}

    \begin{block}{\no\ MusicXML}
      Arbre XML jeràrquic → incompatible amb el flux atòmic del CTC.
    \end{block}
    \vspace{0.2em}
    \begin{block}{\ok\ LMX}
      Tokens plans i atòmics: \code{pitch:C}, \code{quarter}, \code{flat}, \code{measure}.\\
      Vocabulari $\approx$100 tokens · gramàtica regular → corrector determinista.
    \end{block}
  \end{columns}
\end{frame}

% ── 13. Conclusions ────────────────────────────────────────────────────
\begin{frame}{Conclusions i treball futur}

    \begin{enumerate}
      \setlength\itemsep{0.35em}

      \item \textbf{Vocabulari de tresets} — el buit més gran de l'LMX
      \item \textbf{Detector de pentagrames après} (YOLO/DETR) per a escanejos difícils
    \end{enumerate}
  \end{columns}

% ── Backup: LMX gramàtica i exemple ────────────────────────────────────
\begin{frame}{Suport · LMX: gramàtica i exemple}
  \vspace{0.2em}
  \textbf{\color{accent}Ordre dels tokens d'una nota}\\[0.25em]
  {\small \code{pitch:<A--G>} · \code{octave:<0--8>} · \textbf{durada} · \code{dot}* · [alteració] · [\code{tied:stop}] · [\code{fermata}]}
  \vspace{0.5em}
  \begin{itemize}\setlength\itemsep{0.3em}
    \item L'\textbf{alteració} és una paraula simple (\code{flat}/\code{sharp}/\code{natural}) \textbf{després} de la durada; és \textbf{només visual} (el to ja és a \code{pitch})
    \item El \textbf{compàs} són 3 tokens: \code{time} · \code{beats:N} · \code{beat-type:N} → comparteix representació entre mètriques
  \end{itemize}
  \vspace{0.3em}
  \begin{block}{Exemple — 3/4 en Mi\flatt\ major}
    {\footnotesize\ttfamily measure clef:G2 key:fifths:-3 time beats:3 beat-type:4\\ pitch:B octave:5 quarter flat \ \ pitch:E octave:5 eighth flat \ldots}
  \end{block}
\end{frame}

% ── Backup: fuga de dades i correcció ──────────────────────────────────
\begin{frame}{Suport · Higiene de dades: fuga i correcció}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{\color{accent}Què va fallar}
    {\small
    \begin{itemize}\setlength\itemsep{0.3em}
      \item Variants derivades d'una mateixa peça (bessones sense capçalera, re-renders, augmentacions) entraven al conjunt \textbf{abans} de la partició aleatòria
      \item Un quasi-duplicat de \textit{train} podia caure a \textit{val/test} → mètriques inflades
    \end{itemize}}
    \begin{alertblock}{Mètriques pre-correcció — invàlides}
      {\small 0,23\,\% SER · 94,1\,\% perfectes (no comparables).}
    \end{alertblock}

    \column{0.5\textwidth}
    \textbf{\color{accent}La correcció}
    {\small
    \begin{itemize}\setlength\itemsep{0.3em}
      \item Partició \textbf{per id de mostra}: totes les variants (neta + escanejades) al \textbf{mateix costat}
      \item Diversitat \textbf{només a \textit{train}} per construcció — no pot creuar el \textit{split}
      \item Totes les xifres reportades són \textbf{posteriors} a la correcció (run \code{20260612}, època 83)
    \end{itemize}}
    \vspace{0.2em}
    {\footnotesize\color{muted} Detectat i corregit a mig projecte → evidència d'autocorrecció i d'higiene de dades.}
  \end{columns}
\end{frame}

\begin{frame}{Suport · SER brut vs.\ contingut}
